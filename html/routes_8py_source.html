44@router.get("/articles", response_model=PaginatedArticlesResponse)
45def get_articles(
46 db: Session = Depends(get_db),
47 search: Optional[str] = Query(None),
48 page: int = Query(1, ge=1),
49 page_size: int = Query(20, ge=1, le=100)):
50 """Retrieve a paginated list of articles, optionally filtered by search.
51
52 :param db: Injected database session.
53 :param search: Optional search string to filter articles by title.
54 :param page: Page number (1-indexed, default 1).
55 :param page_size: Number of articles per page (default 20, max 100).
56 :returns: Paginated articles response with metadata.
57 :raises: Exception on database or processing errors.
58 """
59 try:
60 offset = (page - 1) * page_size
61
62 if search:
63 logger.info(f"Searching articles with query: {search}, page: {page}, size: {page_size}")
64 else:
65 logger.info(f"Fetching articles - page: {page}, size: {page_size}")
66
67 articles, total_count = get_all_articles(db, search=search, limit=page_size, offset=offset)
68 total_pages = (total_count + page_size - 1) // page_size # Ceiling division
69
70 logger.info(f"Returned {len(articles)} articles ({total_count} total)")
71 return {
72 "items": articles,
73 "total_count": total_count,
74 "page": page,
75 "page_size": page_size,
76 "total_pages": total_pages
77 }
78 except Exception as e:
79 logger.error(f"Error in get_articles: {e}", exc_info=True)
80 raise

134@router.post("/chat/message", response_model=ChatResponse)
135async def send_message(request: ChatRequest):
136 """Send a message in an active chat session and receive an AI response.
137
138 :param request: Chat request containing session ID and user message.
139 :type request: ChatRequest
140 :returns: Chat response with the assistant's reply and updated message history.
141 :rtype: ChatResponse
142 :raises: HTTPException with 404 if session not found, or Exception on processing errors.
143 """
144 try:
145 # Validate session exists
146 session = get_session(request.session_id)
147 if not session:
148 raise HTTPException(status_code=404, detail="Session not found")
149
150 article_id = get_session_article_id(request.session_id)
151
152 # Add user message to history
153 add_message_to_session(request.session_id, "user", request.message)
154
155 # Get all messages in session for context
156 messages = get_session_messages(request.session_id)
157
158 # Call OpenAI with full conversation history
159 response_obj = openai_chat_service(
160 message=request.message,
161 conversation_history=messages,
162 article_id=article_id,
163 session_id=request.session_id
164 )
165
166 # Extract response text from OpenAI API response
167 if isinstance(response_obj, dict):
168 response_text = response_obj.get("response", "") or "I apologize, but I couldn't generate a response."
169 else:
170 response_text = str(response_obj) if response_obj else "I apologize, but I couldn't generate a response."
171
172 # Add assistant response to history
173 add_message_to_session(request.session_id, "assistant", response_text)
174
175 # Return response and updated history
176 updated_messages = get_session_messages(request.session_id)
177 return {
178 "response": response_text,
179 "messages": updated_messages
180 }
181 except HTTPException:
182 raise
183 except Exception as e:
184 logger.error(f"Error processing message: {e}", exc_info=True)
185 raise
