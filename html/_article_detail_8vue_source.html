95<script setup>
96/**
97 * ArticleDetail View Component
98 *
99 * Displays a single article with full content, metadata, and sources.
100 * Renders markdown content as HTML and provides navigation back to the list.
101 * This view serves as the container for the ChatWidget component.
102 *
103 * Features:
104 * - Fetches and displays full article details by ID
105 * - Renders markdown content with custom heading styles
106 * - Displays article metadata (title, date, impact score, tags)
107 * - Shows clickable source links with external link indicators
108 * - Back navigation to article list
109 * - Provides context for the ChatWidget component
110 */
111import { ref, computed, onMounted } from 'vue'
112import { useRoute } from 'vue-router'
113import { apiRequest } from '@/utils/api'
114import { marked } from 'marked';
115
116const route = useRoute()
117
118/** Article data object containing all article information */
119const article = ref({ title: '', content: '', sources: [], tags: [], created_at: null, impact_score: -1 })
120
121/**
122 * Custom markdown renderer for styled headers
123 * Configures heading elements with consistent font weight and sizing
124 */
125const renderer = new marked.Renderer();
126renderer.heading = ({ text, depth }) => {
127 return `<h${depth} style="font-weight: 700; margin-top: 1rem; font-size: 2rem;">${text}</h${depth}>`;
128};
129
130/**
131 * Configure marked library options for GitHub-flavored markdown
132 * Enables line breaks, header IDs, and custom rendering
133 */
134marked.setOptions({
135 breaks: true,
136 gfm: true,
137 headerIds: true,
138 mangle: false,
139 renderer: renderer
140})
141
142/**
143 * Computed property that converts markdown content to HTML
144 * Prior to markdown conversion, adjusts bold-only paragraphs to headings
145 * @returns {string} HTML-rendered article content
146 */
147const renderedContent = computed(() => {
148 if (!article.value.content) return ''
149
150 // Convert bold-only paragraph to real markdown heading
151 // Didn't want to mess with the ai generation side too much
152 let fixed = article.value.content.replace(
153 /^\*\*(.+?)\*\*$/gm,
154 '## $1'
155 )
156
157 return marked(fixed)
158})
159
160/**
161 * Fetch article data when component mounts
162 * Extracts article ID from route params
163 */
164onMounted(async () => {
165 const data = await fetchArticleById(route.params.id)
166 article.value = data || { title: '', content: '', sources: [], tags: [], created_at: null, impact_score: -1 }
167})
168
169/**
170 * Fetch a single article by ID from the API
171 *
172 * @param {string|number} id - The article ID from the route
173 * @returns {Promise<Object>} Article object with all metadata
174 */
175async function fetchArticleById(id) {
176 try {
177 const res = await apiRequest(`/api/articles/${id}`, { method: 'GET' })
178 console.log('Fetched article:', res)
179 return res
180 } catch (e) {
181 console.error('Failed to fetch article:', e.message)
182 return {}
183 }
184}
185
186/**
187 * Format a date string for display with time
188 *
189 * @param {string} dateString - ISO date string from the API
190 * @returns {string} Formatted date like "January 15, 2024 at 10:30 AM"
191 */
192function formatDate(dateString) {
193 const options = { year: 'numeric', month: 'long', day: 'numeric', hour: '2-digit', minute: '2-digit' }
194 return new Date(dateString).toLocaleDateString('en-US', options)
195}
196</script>
