1"""Session management for chat interactions tied to articles.
2
3This module provides in-memory session storage for article-based chat
4conversations. Each session maintains article context (title, content,
5sources, tags, impact score) and conversation history. Sessions are
6identified by UUID and persist until explicitly closed.

32def create_session_with_context(article_id: int, article_title: str, article_content: str, sources: list, tags: list | None = None, impact_score: int | None = None) -> str:
33 """Create a new chat session tied to an article with conversational context.
34
35 Creates a session with an initial greeting message and stores all article
36 metadata (title, content, sources, tags, impact score) for use in AI
37 conversation context. The greeting is added to the message history so
38 it's included in conversation context sent to the AI model.
39
40 :param article_id: ID of the article this session is bound to.
41 :type article_id: int
42 :param article_title: Title of the article.
43 :type article_title: str
44 :param article_content: Full content of the article.
45 :type article_content: str
46 :param sources: List of source dicts associated with the article.
47 :type sources: list
48 :param tags: List of tag dicts (with id and name) associated with the article.
49 :type tags: list or None
50 :param impact_score: Impact score or severity ranking for the article.
51 :type impact_score: int or None
52 :returns: UUID string identifier for the new session.
53 :rtype: str
54 """
55 session_id = str(uuid.uuid4())
56
57 # Create initial greeting message
58 greeting = f'Hello! I\'m here to answer any questions you may have about this article, "{article_title}".'
59
60 SESSIONS[session_id] = {
61 "article_id": article_id,
62 "article_title": article_title,
63 "messages": [
64 {"role": "assistant", "content": greeting}
65 ],
66 "created_at": datetime.now(),
67 "article_content": article_content,
68 "sources": sources or [],
69 "tags": tags or [],
70 "impact_score": impact_score,
71 }
72 return session_id

82def add_message_to_session(session_id: str, role: str, content: str) -> bool:
83 """Add a message to a session's chat history.
84
85 :param session_id: UUID of the session to add the message to.
86 :param role: Message role, typically 'user' or 'assistant'.
87 :param content: Message content/text.
88 :returns: True if message was added, False if session not found.
89 """
90 if session_id not in SESSIONS:
91 return False
92
93 SESSIONS[session_id]["messages"].append({
94 "role": role,
95 "content": content,

99def get_session_messages(session_id: str) -> list:

108def end_session(session_id: str) -> bool:
109 """Terminate a session and clean up.
