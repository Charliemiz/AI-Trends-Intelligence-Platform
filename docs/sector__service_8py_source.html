11# Manages sector rotation state using database storage. State is stored in the database instead of a JSON file.
12class SectorRotationManager:
13 """Manages sector rotation state persisted in the database.
14
15 This class maintains a queue of sectors and tracks which one should be
16 processed next in a round-robin fashion.
17 """
18
19 def __init__(self, db: Session):
20 """Initialize the manager with a database session.
21
22 :param db: Active SQLAlchemy ``Session``.
23 """
24 self.db = db
25 self.state = self._load_state()

27 def _load_state(self) -> dict:
28 """Load rotation state from the database.
29
30 :returns: State dict with current_index, sectors_queue, last_run, and cycle_count.
31 """
32 from backend.db import models
33
34 # Get or create state record
35 state_record = self.db.query(models.SystemState).filter(
36 models.SystemState.key == "sector_rotation"
37 ).first()
38
39 if state_record:
40 import json
41 return json.loads(state_record.value)
42 else:
43 # Create default state
44 return {
45 "current_index": 0,
46 "sectors_queue": [],
47 "last_run": None,
48 "cycle_count": 0
49 }
50 def _save_state(self):
51 """Persist rotation state to the database."""
52 from backend.db import models
53 import json
54
55 state_record = self.db.query(models.SystemState).filter(
56 models.SystemState.key == "sector_rotation"
57 ).first()
58
59 if state_record:
60 state_record.value = json.dumps(self.state)
61 state_record.updated_at = datetime.now()
62 else:
63 state_record = models.SystemState(
64 key="sector_rotation",
65 value=json.dumps(self.state)
66 )
67 self.db.add(state_record)
68
69 self.db.commit()

71 def initialize_sectors(self, sectors: list):
72 """Initialize or reset the sectors queue.
73
74 :param sectors: List of sector names to cycle through.
75 """
76 if not self.state["sectors_queue"] or set(sectors) != set(self.state["sectors_queue"]):
77 self.state["sectors_queue"] = sectors.copy()
78 self.state["current_index"] = 0
79 self._save_state()
80
81 def get_next_sectors(self) -> Optional[str]:
82 """Get the next sector in rotation and advance the pointer.
83
84 :returns: The next sector name, or ``None`` if no sectors configured.
85 """
86 if not self.state["sectors_queue"]:
87 return None
88
89 # Get current sector
90 sector = self.state["sectors_queue"][self.state["current_index"]]
91
92 # Move to next index
93 self.state["current_index"] += 1
94
95 # Reset if we've gone through all sectors
96 if self.state["current_index"] >= len(self.state["sectors_queue"]):
97 self.state["current_index"] = 0
98 self.state["cycle_count"] += 1
99
100 # Update last run time
101 self.state["last_run"] = datetime.now().isoformat()
102
103 self._save_state()
104
105 return sector
106
107 # Get current rotation state
108 def get_current_state(self) -> dict:
109 """Get current rotation state information.
110
111 :returns: Dict with current_sector, sectors_remaining, total_sectors, cycle_count, last_run.
112 """
113 return {
114 "current_sector": self.state["sectors_queue"][self.state["current_index"]]
115 if self.state["sectors_queue"] else None,
116 "sectors_remaining": len(self.state["sectors_queue"]) - self.state["current_index"],
117 "total_sectors": len(self.state["sectors_queue"]),
118 "cycle_count": self.state["cycle_count"],
119 "last_run": self.state["last_run"]
120 }
121
122 # Reset rotation to beginning
123 def reset(self):
124 """Reset rotation to the beginning (index 0, cycle_count 0)."""
125 self.state["current_index"] = 0
126 self.state["cycle_count"] = 0
127 self._save_state()
