85<script setup>
86/**
87 * ChatWidget Component
88 *
89 * A collapsible chat interface that appears on article detail pages.
90 * Manages chat sessions, conversation history, and real-time messaging
91 * with an AI analyst focused on the current article's content.
92 *
93 * Features:
94 * - Auto-creates chat session when navigating to an article page
95 * - Auto-closes session when navigating away
96 * - Maintains conversation history tied to backend session
97 * - Displays initial greeting with article title
98 * - Handles loading states during session creation and messaging
99 */
100import { ref, onMounted, computed, watch } from "vue";
101import { useRoute } from "vue-router";
102import { apiRequest } from "@/utils/api";
103
104const route = useRoute();
105
106/** Whether the chat widget is expanded or collapsed */
107const isOpen = ref(false);
108
109/** Current user input in the message field */
110const input = ref("");
111
112/** Array of message objects with role ('user' or 'assistant') and content */
113const messages = ref([]);
114
115/** Current session ID (UUID) for backend session tracking */
116const sessionId = ref(null);
117
118/** Whether a message is currently being sent */
119const isLoading = ref(false);
120
121/** Whether a session is currently being created */
122const sessionCreating = ref(false);
123
124/**
125 * Check if we're currently on an article detail page
126 * @returns {boolean} True if current route matches /articles/:id pattern
127 */
128const isArticleDetail = computed(() => {
129 // More reliable check - just check if URL matches pattern
130 const matches = route.path.match(/^\/articles\/\d+$/);
131 return !!matches;
132});
133
134/**
135 * Extract article ID from the current route
136 * @returns {number|null} Article ID if on article page, null otherwise
137 */
138const articleId = computed(() => {
139 const match = route.path.match(/^\/articles\/(\d+)$/);
140 return match ? parseInt(match[1], 10) : null;
141});
142
143/**
144 * Watch for route changes to manage session lifecycle
145 * - Closes session when leaving an article page
146 * - Creates session when entering an article page
147 */
148watch([isArticleDetail, articleId], async () => {
149 // If we're leaving an article page, close the session
150 if (!isArticleDetail.value && sessionId.value) {
151 await closeSession();
152 return;
153 }
154
155 // If we're entering an article page, create a new session
156 if (isArticleDetail.value && articleId.value && !sessionId.value) {
157 await createSession(articleId.value);
158 }
159});
160
161/**
162 * Create a chat session on component mount if on an article page
163 * Ensures session is ready when user first visits an article
164 */
165onMounted(async () => {
166 if (isArticleDetail.value && articleId.value && !sessionId.value) {
167 await createSession(articleId.value);
168 }
169});
170
171/**
172 * Create a chat session and initialize with article context
173 *
174 * Fetches the full article data and sends it to the backend to create
175 * a session. The backend returns the session ID and initial greeting
176 * message which is displayed in the chat.
177 *
178 * @param {number} id - The article ID to create a session for
179 */
180const createSession = async (id) => {
181 sessionCreating.value = true;
182 try {
183 // Fetch article for context
184 const article = await apiRequest(`/api/articles/${id}`, { method: "GET" });
185 const response = await apiRequest("/api/chat/session", {
186 method: "POST",
187 body: JSON.stringify({
188 article_id: id,
189 article_title: article?.title || "",
190 article_content: article?.content || "",
191 sources: article?.sources || [],
192 tags: article?.tags || [],
193 impact_score: article?.impact_score || null
194 })
195 });
196 sessionId.value = response.session_id;
197 // Use the initial messages (greeting) returned from the backend
198 messages.value = response.messages || [];
199 console.log(`Chat session created: ${sessionId.value} for article ${id}`);
200 } catch (err) {
201 console.error("Failed to create chat session:", err.message);
202 console.error("Error details:", err);
203 // Set fallback greeting on error
204 messages.value = [
205 {
206 role: "assistant",
207 content: "Hello! I'm here to answer any questions about this article."
208 }
209 ];
210 } finally {
211 sessionCreating.value = false;
212 }
213};
214
215/**
216 * Close the current chat session and clear local state
217 *
218 * Sends a DELETE request to the backend to terminate the session
219 * and remove it from memory. Clears the local session ID and messages.
220 */
221const closeSession = async () => {
222 if (sessionId.value) {
223 try {
224 await apiRequest(`/api/chat/session/${sessionId.value}`, {
225 method: "DELETE"
226 });
227 console.log(`Chat session closed: ${sessionId.value}`);
228 } catch (err) {
229 console.error("Failed to close session:", err.message);
230 }
231 sessionId.value = null;
232 messages.value = [];
233 }
234};
235
236/**
237 * Send a user message to the chat backend and display the response
238 *
239 * Validates input, adds the user message to the UI immediately,
240 * sends it to the backend with the session ID, and appends the
241 * AI response to the conversation when received.
242 */
243const sendMessage = async () => {
244 if (!input.value.trim() || !sessionId.value) return;
245
246 // Add user message to chat
247 messages.value.push({ role: "user", content: input.value });
248
249 const userInput = input.value;
250 input.value = "";
251 isLoading.value = true;
252
253 try {
254 const res = await apiRequest("/api/chat/message", {
255 method: "POST",
256 body: JSON.stringify({
257 session_id: sessionId.value,
258 message: userInput
259 })
260 });
261 const reply = res.response;
262 messages.value.push({ role: "assistant", content: reply });
263 } catch (err) {
264 console.error("Chat error:", err.message);
265 messages.value.push({
266 role: "assistant",
267 content: "Sorry, there was an error reaching the analyst.",
268 });
269 } finally {
270 isLoading.value = false;
271 }
272};
